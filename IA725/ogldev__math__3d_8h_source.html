272 m[1][0] = AssimpMatrix.b1; m[1][1] = AssimpMatrix.b2; m[1][2] = AssimpMatrix.b3; m[1][3] = 0.0f;
273 m[2][0] = AssimpMatrix.c1; m[2][1] = AssimpMatrix.c2; m[2][2] = AssimpMatrix.c3; m[2][3] = 0.0f;

304 inline Matrix4f operator*(const Matrix4f& Right) const
305 {
306 Matrix4f Ret;
307
308 for (unsigned int i = 0 ; i < 4 ; i++) {
309 for (unsigned int j = 0 ; j < 4 ; j++) {
310 Ret.m[i][j] = m[i][0] * Right.m[0][j] +
311 m[i][1] * Right.m[1][j] +
312 m[i][2] * Right.m[2][j] +

382 {

385 float z = this->pos.z;
